## 🚀 Getting Started

### Installation (Future)
```bash
npm install @dynui-max/core @dynui-max/design-tokens
# or
pnpm add @dynui-max/core @dynui-max/design-tokens
```

### Basic Usage
```tsx
import { 
  ThemeProvider,
  DynContainer,
  DynBox,
  DynButton, 
  DynInput,
  DynSelect
} from '@dynui-max/core';
import '@dynui-max/design-tokens/css';

function App() {
  return (
    <ThemeProvider defaultTheme="light">
      <DynContainer size="lg">
        <DynBox p="lg" bg="primary" radius="lg">
          <h1>Welcome to DynUI-Max</h1>
          <DynBox display="flex" gap="md">
            <DynInput placeholder="Enter your name..." clearable />
            <DynButton variant="solid" color="primary">Submit</DynButton>
          </DynBox>
        </DynBox>
      </DynContainer>
    </ThemeProvider>
  );
}
```
